CREATE DATABASE GymManagement;
GO

USE GymManagement;
GO
CREATE TABLE VaiTro (
    MaVaiTro INT PRIMARY KEY IDENTITY,
    TenVaiTro NVARCHAR(50) -- Admin, Staff
);

CREATE TABLE TaiKhoan (
    MaTaiKhoan INT PRIMARY KEY IDENTITY,
    Username NVARCHAR(50) UNIQUE,
    Password NVARCHAR(100),
    MaVaiTro INT,
    TrangThai BIT DEFAULT 1,
    FOREIGN KEY (MaVaiTro) REFERENCES VaiTro(MaVaiTro)
);
CREATE TABLE HoiVien (
    MaHoiVien INT PRIMARY KEY IDENTITY,
    Ten NVARCHAR(100),
    NgaySinh DATE,
    GioiTinh NVARCHAR(10),
    SoDienThoai NVARCHAR(15),
    DiaChi NVARCHAR(255),
    NgayDangKy DATE DEFAULT GETDATE()
);
CREATE TABLE PT (
    MaPT INT PRIMARY KEY IDENTITY,
    Ten NVARCHAR(100),
    SoDienThoai NVARCHAR(15),
    KinhNghiem NVARCHAR(255),
    TrangThai BIT DEFAULT 1
);
CREATE TABLE GoiTap (
    MaGoi INT PRIMARY KEY IDENTITY,
    TenGoi NVARCHAR(100),
    Gia DECIMAL(10,2),
    ThoiHan INT, -- số ngày
    MoTa NVARCHAR(255)
);
CREATE TABLE DangKyGoi (
    MaDangKy INT PRIMARY KEY IDENTITY,
    MaHoiVien INT,
    MaGoi INT,
    NgayBatDau DATE,
    NgayKetThuc DATE,
    TrangThai NVARCHAR(50),
    FOREIGN KEY (MaHoiVien) REFERENCES HoiVien(MaHoiVien),
    FOREIGN KEY (MaGoi) REFERENCES GoiTap(MaGoi)
);
CREATE TABLE SanPham (
    MaSanPham INT PRIMARY KEY IDENTITY,
    TenSanPham NVARCHAR(100),
    Gia DECIMAL(10,2),
    SoLuong INT,
    MoTa NVARCHAR(255)
);
CREATE TABLE HoaDon (
    MaHoaDon INT PRIMARY KEY IDENTITY,
    NgayLap DATETIME DEFAULT GETDATE(),
    MaTaiKhoan INT,
    TongTien DECIMAL(10,2),
    FOREIGN KEY (MaTaiKhoan) REFERENCES TaiKhoan(MaTaiKhoan)
);
CREATE TABLE ChiTietHoaDon (
    MaChiTiet INT PRIMARY KEY IDENTITY,
    MaHoaDon INT,
    MaSanPham INT,
    SoLuong INT,
    DonGia DECIMAL(10,2),
    FOREIGN KEY (MaHoaDon) REFERENCES HoaDon(MaHoaDon),
    FOREIGN KEY (MaSanPham) REFERENCES SanPham(MaSanPham)
);
CREATE TABLE LichTap (
    MaLich INT PRIMARY KEY IDENTITY,
    MaHoiVien INT,
    MaPT INT,
    NgayTap DATE,
    GioTap TIME,
    TrangThai NVARCHAR(50),
    FOREIGN KEY (MaHoiVien) REFERENCES HoiVien(MaHoiVien),
    FOREIGN KEY (MaPT) REFERENCES PT(MaPT)
);
CREATE TABLE CaLam (
    MaCa INT PRIMARY KEY IDENTITY,
    TenCa NVARCHAR(50),
    GioBatDau TIME,
    GioKetThuc TIME
);
CREATE TABLE DangKyCa (
    MaDangKyCa INT PRIMARY KEY IDENTITY,
    MaTaiKhoan INT,
    MaCa INT,
    NgayLam DATE,
    FOREIGN KEY (MaTaiKhoan) REFERENCES TaiKhoan(MaTaiKhoan),
    FOREIGN KEY (MaCa) REFERENCES CaLam(MaCa)
);
